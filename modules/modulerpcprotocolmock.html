getMockRpcClient

  • getMockRpcClient<serviceDefinition, ResponseContext>(serviceDefinition: serviceDefinition, streamProducer: StreamProducer): Service<serviceDefinition, ResponseContext>
    • Defined in protocol/mock/mock.ts:38

    Mocks an RPC client, calling a stream producer that can be used to mock server responses.

    example
    const serviceDefinition = {
  foo: { request: {}, response: {} as { bar: number } },
};
const client = getMockRpcClient(serviceDefinition, (method, request) => {
  console.log(
    'RPC:',
    method,
    request instanceof Function ? request() : request,
  );
  return ModuleRpcClient.streamFromArray([
    { response: { bar: 1 } },
  ]) as ModuleRpcClient.Stream<any>;
});
const { bar } = await client.nice().foo({});
expect(bar).to.equal(1);
